NEW SYMBOL Round_First_Action
	WHEN (NOT didrais AND NOT didcall AND NOT didchec) RETURN 1 FORCE 
	WHEN (OTHERS) RETURN 0 FORCE
END SYMBOL

NEW SYMBOL PostFlop_Senario  
// 1. Eu sou o primeiro a agir 
// 2. Viloes checaram sou o ultimo a agir
// 3. Viloes checaram nao sou o ultimo a agir
// 4. Vilao fez Raise
// 5. Eu fiz raise e recebi ReRaise
	WHEN (Round_First_Action = 1 AND betposition = 1) RETURN 1 FORCE
	WHEN (Round_First_Action = 1 AND nbetstocall =0 AND (betposition = nplayersround)) RETURN 2 FORCE
	WHEN (Round_First_Action = 1 AND nbetstocall =0 AND betposition!=1 AND (betposition != nplayersround)) RETURN 3 FORCE
	WHEN (didrais AND nbetstocall > 0) RETURN 5 FORCE
	WHEN (nbetstocall > 0) RETURN 4 FORCE
END SYMBOL
